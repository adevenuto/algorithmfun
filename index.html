function  fizzBuzz(maxNum) {
    for (i=1;i<=maxNum;i++) {
        if (i % 15 === 0) {
            console.log('Fizbuzz');
        } else if (i % 5 === 0) {
            console.log('Buzz');
        } else if (i % 3 === 0) {
            console.log('Fizz');
        } else {
            console.log(i);
        }
    }
}
